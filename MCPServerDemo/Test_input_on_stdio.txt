  "jsonrpc": "2.0",
  "id": "1",
  "result": {
    "tools": [
      {
        "name": "get_time",
        "description": "Returns current server time",
        "input_schema": {
          "type": "object",
          "properties": {
            "Format": {

            }

        "name": "calculate_sum",

              "type": "integer",
